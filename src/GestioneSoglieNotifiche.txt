App.jsx
import { useState } from 'react';
import Header from './components/Header';
import Card from './components/Card';
import StepperControl from './components/StepperControl';
import ToggleSwitch from './components/ToggleSwitch';
import BottomNav from './components/BottomNav';

export default function App() {
  const [tempMinima, setTempMinima] = useState(18);
  const [tempMassima, setTempMassima] = useState(25);
  const [batteriaCritica, setBatteriaCritica] = useState(20);
  const [pesoMassimo, setPesoMassimo] = useState(50);
  const [notifichePush, setNotifichePush] = useState(true);

  return (
    // Sfondo marrone molto scuro / nero
    <div className="min-h-screen bg-[#0f0e0c] pb-10 font-sans">
      
      {/* Header personalizzato con colori oro */}
      <header className="py-8 px-5 border-b border-[#ffcc00]/30">
        <h1 className="text-3xl font-bold text-white text-center">
          ARNIA <span className="text-[#ffcc00] font-light">[IMPOSTAZIONI]</span>
        </h1>
        <div className="w-24 h-1 bg-[#ffcc00] mx-auto mt-2 rounded-full shadow-[0_0_8px_#ffcc00]"></div>
      </header>

      <div className="max-w-md mx-auto px-5 mt-6">
        
        <Card title="Soglia Temperatura">
          <div className="flex gap-4">
            <StepperControl label="Min" value={tempMinima} unit="°C" onDecrease={() => setTempMinima(tempMinima - 1)} onIncrease={() => setTempMinima(tempMinima + 1)} />
            <StepperControl label="Max" value={tempMassima} unit="°C" onDecrease={() => setTempMassima(tempMassima - 1)} onIncrease={() => setTempMassima(tempMassima + 1)} />
          </div>
        </Card>

        <Card title="Batteria">
          <StepperControl label="Soglia Critica" value={batteriaCritica} unit="%" onDecrease={() => setBatteriaCritica(batteriaCritica - 1)} onIncrease={() => setBatteriaCritica(batteriaCritica + 1)} />
        </Card>

        <Card title="Peso Massimo" subtitle="Valore soglia avviso">
          <StepperControl label="Peso" value={pesoMassimo} unit=" Kg" onDecrease={() => setPesoMassimo(pesoMassimo - 1)} onIncrease={() => setPesoMassimo(pesoMassimo + 1)} />
        </Card>

        <Card title="Notifiche">
          <ToggleSwitch label="Stato Avvisi" isActive={notifichePush} onToggle={setNotifichePush} />
        </Card>

        <BottomNav />

      </div>
    </div>
  );
}


Card.jsx
export default function Card({ title, subtitle, children }) {
  return (
    <div className="bg-[#1a1a1a] border border-[#ffcc00] rounded-2xl p-5 mb-4 shadow-[0_0_10px_rgba(255,204,0,0.2)]">
      <h2 className="text-lg font-semibold text-[#ffcc00] mb-1 uppercase tracking-wider">{title}</h2>
      {subtitle && <p className="text-xs text-gray-400 mb-3 italic">{subtitle}</p>}
      <div className="mt-2 text-white">
        {children}
      </div>
    </div>
  );
}


StepperControl.jsx
export default function StepperControl({ label, value, unit, onIncrease, onDecrease }) {
  return (
    <div className="flex-1">
      <label className="block text-xs text-gray-400 mb-2 uppercase">{label}</label>
      <div className="flex items-center gap-2 bg-black border border-gray-700 px-3 py-2 rounded-lg">
        <button 
          onClick={onDecrease}
          className="text-[#ffcc00] text-xl font-bold px-2 hover:bg-[#ffcc00] hover:text-black rounded transition-colors"
        >
          -
        </button>
        <span className="flex-1 text-center text-white font-medium text-sm">
          {value}{unit}
        </span>
        <button 
          onClick={onIncrease}
          className="text-[#ffcc00] text-xl font-bold px-2 hover:bg-[#ffcc00] hover:text-black rounded transition-colors"
        >
          +
        </button>
      </div>
    </div>
  );
}


ToggleSwitch.jsx
export default function ToggleSwitch({ label, isActive, onToggle }) {
  return (
    <div className="flex justify-between items-center">
      <label className="text-xs text-gray-400 uppercase">{label}</label>
      <div className="flex bg-black border border-[#ffcc00] p-1 rounded-xl">
        <button 
          onClick={() => onToggle(true)}
          className={`px-4 py-1 rounded-lg text-xs font-bold transition-all ${
            isActive ? 'bg-[#ffcc00] text-black' : 'text-gray-500'
          }`}
        >
          ON
        </button>
        <button 
          onClick={() => onToggle(false)}
          className={`px-4 py-1 rounded-lg text-xs font-bold transition-all ${
            !isActive ? 'bg-[#ffcc00] text-black' : 'text-gray-500'
          }`}
        >
          OFF
        </button>
      </div>
    </div>
  );
}


Header.jsx
// components/Header.jsx
export default function Header({ title }) {
  return (
    <header className="bg-gray-800 text-white text-center py-5 rounded-b-3xl mb-5">
      <h1 className="text-2xl font-medium">{title}</h1>
    </header>
  );
}


BottomNav.jsx
export default function BottomNav() {
  return (
    <div className="flex gap-3 mt-8">
      <button className="flex-1 py-3 rounded-xl font-bold text-xs border border-[#ffcc00] text-[#ffcc00] bg-transparent hover:bg-[#ffcc00]/10 transition-all uppercase">
        HOME
      </button>
      <button className="flex-1 py-3 rounded-xl font-bold text-xs border border-[#ffcc00] text-[#ffcc00] bg-transparent hover:bg-[#ffcc00]/10 transition-all uppercase">
        IMPOSTAZIONI
      </button>
    </div>
  );
}
